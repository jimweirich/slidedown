Syntax Highlighting

  @@@ ruby
      # A rack app
      proc {
        [200, { 'Location' => '/' },
          "You are being redirected to /"
        ]
      }
  @@@

Becomes...

# A rack app
proc {
  [200, { 'Location' => '/' },
    "You are being redirected to /"
  ]
}

  @@@ js
      // Some JavaScript
      function() {
        if (1 == 2) {
          return "This is silly."
        }
      }
  @@@

Becomes...

// Some JavaScript
function() {
  if (1 == 2) {
    return "This is silly."
  }
}
